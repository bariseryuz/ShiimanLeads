            try {
              await page.goto(source.url, navOpts);
              logger.info(`✅ Loaded Phoenix page`);
              
              await page.waitForSelector('select', { timeout: 10000 });
              await new Promise(r => setTimeout(r, 2000));
              
              // STEP 1: Select "-ALL BUILDING PERMITS-"
              logger.info('Step 1: ALL BUILDING PERMITS');
              await page.evaluate(() => {
                const sel = document.querySelector('select[name="PermitType"]');
                if (sel) {
                  sel.value = '-ALL BUILDING PERMITS-';
                  sel.dispatchEvent(new Event('change', { bubbles: true }));
                }
              });
              await new Promise(r => setTimeout(r, 1000));
              
              // STEP 2: Select "007 - 10 OR MORE FAMILY UNITS"
              logger.info('Step 2: 10 OR MORE FAMILY UNITS');
              await page.evaluate(() => {
                const sel = document.querySelector('select[name="StructureClass"]');
                if (sel) {
                  sel.value = '007 - 10 OR MORE FAMILY UNITS';
                  sel.dispatchEvent(new Event('change', { bubbles: true }));
                }
              });
              await new Promise(r => setTimeout(r, 1000));
              
              // STEP 3: Enter date 365 days ago
              logger.info('Step 3: Enter start date (365 days ago)');
              const startDate = new Date();
              startDate.setDate(startDate.getDate() - 365);
              const dateStr = `${startDate.getMonth() + 1}/${startDate.getDate()}/${startDate.getFullYear()}`;
              await page.evaluate((d) => {
                const inp = document.querySelector('input[name="txtStartDate"]');
                if (inp) {
                  inp.value = d;
                  inp.dispatchEvent(new Event('input', { bubbles: true }));
                }
              }, dateStr);
              await new Promise(r => setTimeout(r, 1000));
              
              // STEP 4: Select "Struct Class" radio button
              logger.info('Step 4: Sort by Struct Class');
              await page.evaluate(() => {
                const radio = document.querySelector('input[type="radio"][value="Struct Class"]');
                if (radio) {
                  radio.checked = true;
                  radio.dispatchEvent(new Event('change', { bubbles: true }));
                }
              });
              await new Promise(r => setTimeout(r, 1000));
              
              // STEP 5: Click "Create List"
              logger.info('Step 5: Click Create List button');
              await page.evaluate(() => {
                const btn = document.querySelector('input[value="Create List"]');
                if (btn) btn.click();
              });
              
              logger.info('⏳ Waiting for results...');
              await new Promise(r => setTimeout(r, 10000));
              await page.waitForSelector('table', { timeout: 20000 });
              logger.info('✅ Results loaded!');
              
              // Loop: Screenshot → AI extract → Click next
              let pageNum = 1;
              let hasNext = true;
              
              while (hasNext && pageNum <= 50) {
                logger.info(`📸 Page ${pageNum}: Taking screenshot`);
                const screenshot = await page.screenshot({ fullPage: true });
                
                logger.info(`🤖 Page ${pageNum}: AI extracting`);
                const leads = await extractLeadWithAI(
                  screenshot.toString('base64'),
                  source.name,
                  source.fieldSchema || {},
                  true,
                  'image/png'
                );
                
                if (leads && leads.length > 0) {
                  logger.info(`✅ Page ${pageNum}: ${leads.length} leads extracted`);
                  
                  for (const lead of leads) {
                    const inserted = await insertLeadIfNew({
                      raw: JSON.stringify(lead),
                      sourceName: source.name,
                      lead: lead,
                      userId: userId,
                      sourceId: source._sourceId || source.id,
                      extractedData: lead
                    });
                    if (inserted) newLeads++;
                  }
                }
                
                // Puppeteer: Click Next
                const clicked = await page.evaluate(() => {
                  const btns = Array.from(document.querySelectorAll('a, button, input'));
                  const next = btns.find(b => {
                    const t = (b.innerText || b.value || '').toLowerCase().trim();
                    return t === 'next' || t === '>' || t === '›';
                  });
                  if (next && !next.disabled && next.offsetParent) {
                    next.click();
                    return true;
                  }
                  return false;
                });
                
                if (clicked) {
                  logger.info(`➡️ Page ${pageNum}: Clicked Next`);
                  await new Promise(r => setTimeout(r, 5000));
                  try {
                    await page.waitForFunction(() => document.querySelector('table tr').length > 1, { timeout: 10000 });
                    pageNum++;
                  } catch {
                    hasNext = false;
                  }
                } else {
                  logger.info('✓ No more pages');
                  hasNext = false;
                }
              }
              
              logger.info(`✅ Phoenix done: ${pageNum} pages, ${newLeads} new leads`);
              
              if (browser) await browser.close();
              await updateSourceStatus(source._sourceId || source.id, 'success', new Date());
              await updateProgress(userId, { newLeads });
              continue;
              
            } catch (err) {
              logger.error(`❌ Phoenix failed: ${err.message}`);
              if (browser) await browser.close();
              continue;
            }
